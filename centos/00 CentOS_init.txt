
配置网卡
如果虚拟机复制后，选择“I copied it.”选项后，需要做如下修正网卡配置，才能启动网卡：
# vi /etc/udev/rules.d/70-persistent-net.rules 
SUBSYSTEM=="net", ACTION=="add", DRIVERS=="?*", ATTR{address}=="00:0c:29:66:3f:1f", ATTR{type}=="1", KERNEL=="eth*", NAME="eth0" 
:wq
将原eth0配置删除，将eth1改为eth0，同时记录下MAC地址：00:0c:29:66:3f:1f备用。

# vi /etc/sysconfig/network-scripts/ifcfg-eth0 
DEVICE="eth0"
BOOTPROTO="static"
GATEWAY="192.168.203.2"
NETMASK="255.255.255.0"
IPADDR="192.168.203.80"
DNS1="192.168.203.2"
DNS2="8.8.8.8"
NM_CONTROLLED="yes"
ONBOOT=yes
TYPE="Ethernet"
DEFROUTE=yes
IPV4_FAILURE_FATAL=yes
IPV6INIT=no
IPV6_AUTOCONF=no
NAME="System eth0"
HWADDR=00:0C:29:66:3F:1F
PEERDNS=yes
PEERROUTES=yes
:wq 
# service network restart 

配置hostname
vi /etc/sysconfig/network
NETWORKING=yes　　　　　　
HOSTNAME=web1
:wq
hostname web1        #立即生效

禁用selinux
vi /etc/sysconfig/selinux
SELINUX=disabled
#SELINUXTYPE=targeted
:wq
# reboot

ntpdate time.nist.gov
date
写入bios
hwclock -w

定时校正服务器时钟，定时与中国国家授时中心授时服务器同步
echo '*/30 * * * *     /usr/sbin/ntpdate time.nist.gov > /root/ntpdate.log 2>&1' >> /etc/crontab


添加用户
useradd capvision
passwd capvision
chmod 740 /etc/sudoers
vi /etc/sudoers
%wheel  ALL=(ALL)       ALL
:wq
chmod 440 /etc/sudoers
usermod -g root capvision
su capvision

禁止root本地登录
vi /etc/pam.d/login
# shift+G, o
auth required pam_succeed_if.so user != root quiet
:wq

禁止root远程ssh登录
vi /etc/ssh/sshd_config
PermitRootLogin no
:wq
service sshd restart

删除用户（删除前先备份/home/www目录）
userdel -rf www
