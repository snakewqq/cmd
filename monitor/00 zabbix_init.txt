安装zabbix官方源和epel源
rpm -ivh http://repo.zabbix.com/zabbix/2.2/rhel/6/x86_64/zabbix-release-2.2-1.el6.noarch.rpm
rpm -ivh http://dl.fedoraproject.org/pub/epel/6/x86_64/epel-release-6-8.noarch.rpm

put zabbix-release-2.2-1.el6.noarch.rpm epel-release-6-8.noarch.rpm

yum -y install zabbix zabbix-get zabbix-server zabbix-web-mysql zabbix-web zabbix-agent

vi /etc/my.cnf
innodb_file_per_table
symbolic-links=0
character-set-server=utf8
:wq

service mysqld restart
mysql -uroot -p
mysql> create database zabbix character set utf8;
mysql> use zabbix;
mysql> source /usr/share/doc/zabbix-server-mysql-2.2.10/create/schema.sql;
mysql> source /usr/share/doc/zabbix-server-mysql-2.2.10/create/images.sql;
mysql> source /usr/share/doc/zabbix-server-mysql-2.2.10/create/data.sql;
mysql> show tables;
mysql> exit;

配置zabbix
egrep -v "(^#|^$)" /etc/zabbix/zabbix_server.conf
vim /etc/zabbix/zabbix_server.conf
DBHost=127.0.0.1
DBUser=admin
DBPassword=admin1234
CacheSize=256M
:wq

mkdir -p /etc/zabbix/alertscripts
mkdir -p /etc/zabbix/externalscripts

vi /etc/sysconfig/iptables
-A INPUT -m state --state NEW -m tcp -p tcp --dport 10050:10051 -j ACCEPT
:wq
service iptables restart

启动zabbix
service zabbix-server start
service httpd start

启动浏览器：
http://192.168.5.87/zabbix
按提示一步一步安装

安装成功后，使用如下账户登录后台：
账户：admin
密码：zabbix
